/* * ============================================================ * MÓDULO: NAVEGACIÓN * * Gestiona el sidebar y la navegación tipo SPA (Single Page * Application): muestra/oculta secciones sin recargar la página. * ============================================================ */ /* Referencias DOM del sidebar */ const sidebar = document.getElementById('sidebar'); const sidebarOverlay = document.getElementById('sidebarOverlay'); const menuToggle = document.getElementById('menuToggle'); const sidebarClose = document.getElementById('sidebarClose'); /** Abre el sidebar y activa el overlay (comportamiento móvil) */ function openSidebar() { sidebar.classList.add('open'); sidebarOverlay.classList.add('show'); } /** Cierra el sidebar y desactiva el overlay */ function closeSidebar() { sidebar.classList.remove('open'); sidebarOverlay.classList.remove('show'); } menuToggle.addEventListener('click', openSidebar); sidebarClose.addEventListener('click', closeSidebar); sidebarOverlay.addEventListener('click', closeSidebar); /* clic fuera cierra el menú */ /* * Mapa de nombres de sección → título del topbar. * Se actualiza cada vez que el usuario navega a una nueva página. */ const pageNames = { 'dashboard' : 'Panel de Control', 'inventory' : 'Inventario', 'add-product' : 'Agregar Producto', 'invoice' : 'Facturación', 'reports' : 'Reportes Generales', 'sales-reports' : 'Reportes de Ventas', 'maintenance' : 'Mantenimiento', 'credits' : 'Créditos', 'credits-reports' : 'Reportes de Créditos', }; /** * Navega a la sección indicada. * * Pasos: * 1. Oculta todas las páginas (.page → .hidden) * 2. Quita la clase active de todos los ítems del nav * 3. Muestra la página destino * 4. Marca el ítem del nav correspondiente como activo * 5. Actualiza el título del topbar * 6. Cierra el sidebar en móvil * 7. Notifica a los módulos JS si showTab() está disponible * * @param {string} tabId - ID del div.page a mostrar */ function navigateTo(tabId) { // Ocultar TODAS las páginas — quitar hidden Y active para evitar estados mixtos document.querySelectorAll('.page').forEach(p => { p.classList.add('hidden'); p.classList.remove('active'); }); document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active')); // Mostrar solo la página destino const page = document.getElementById(tabId); if (page) { page.classList.remove('hidden'); page.classList.add('active'); } const navItem = document.querySelector(`.nav-item[data-tab="${tabId}"]`); if (navItem) navItem.classList.add('active'); document.getElementById('topbarTitle').textContent = pageNames[tabId] || tabId; closeSidebar(); if (typeof showTab === 'function') showTab(tabId); } /* Registrar navegación en ítems del sidebar */ document.querySelectorAll('.nav-item[data-tab]').forEach(item => { item.addEventListener('click', () => navigateTo(item.dataset.tab)); }); /* Registrar navegación en tarjetas del dashboard */ document.querySelectorAll('.dash-card[data-goto]').forEach(card => { card.addEventListener('click', () => navigateTo(card.dataset.goto)); }); /* * Atajo de teclado F2: * Navega a Facturación si el sistema principal está activo. * Útil para agilizar el flujo de venta en caja. */ document.addEventListener('keydown', e => { const sistemaActivo = document.getElementById('mainSystem'); if (e.key === 'F2' && sistemaActivo && !sistemaActivo.classList.contains('hidden')) { navigateTo('invoice'); } });
